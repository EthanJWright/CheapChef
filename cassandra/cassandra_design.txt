  WITH REPLICATION = { 'class' : 'SimpleStrategy', 'replication_factor' : 3 };

CREATE TABLE ingredients(
    ingredient int PRIMARY KEY,
    ingredient_name text,
    ingredient_price int,
    category text,
    );


CREATE TABLE categories(
  ingredient int PRIMARY KEY,
  ingredient_name text,
  ingredient_price int,
  category text,
  );
